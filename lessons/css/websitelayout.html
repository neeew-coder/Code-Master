body {
  display: grid;
  grid-template-areas:
    "header header"
    "nav nav"
    "main aside"
    "footer footer";
  grid-template-columns: 2fr 1fr;
  gap: 16px;
  padding: 20px;
}

header    { grid-area: header; background: #1e3a8a; color: white; padding: 12px; }

article   { grid-area: main; background: white; padding: 12px; }
aside     { grid-area: aside; background: #f0f9ff; padding: 12px; }
footer    { grid-area: footer; background: #1e3a8a; color: white; text-align: center; padding: 12px; }

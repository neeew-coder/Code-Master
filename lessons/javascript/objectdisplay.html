üìÑ Display Techniques

let user = {
  name: "User",
  age: 25,
  isStudent: true
};

// Loop through keys
for (let key in user) {
  console.log(key + ": " + user[key]);
}

// Get keys and values
console.log(Object.keys(user));   // ["name", "age", "isStudent"]
console.log(Object.values(user)); // ["User", 25, true]

// Convert to string
console.log(JSON.stringify(user)); // '{"name":"User","age":25,"isStudent":true}'

Use for...in for iteration, Object.keys() and Object.values() for arrays, and JSON.stringify() for readable output.

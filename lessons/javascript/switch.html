üìÑ Syntax

let message;

  case "Monday":
    message = "Start of the week";
    break;
  case "Friday":
    message = "Almost weekend";
    break;
  case "Sunday":
    message = "Rest day";
    break;
  default:
    message = "Just another day";
}

using System;

class Animal {
  public virtual void Speak() {
    Console.WriteLine("Animal makes a sound");
  }
}

class Cat : Animal {
  public override void Speak() {
    Console.WriteLine("Cat meows");
  }
}

class Program {
  static void Main() {
    Animal a = new Cat();
    a.Speak(); // Output: Cat meows
  }
}

using System;

class Program {
  static void Main() {
    int a = 5;
    int b = 0;

    try {
      int result = Divide(a, b); // Set breakpoint here
      Console.WriteLine("Result: " + result);
    } catch (DivideByZeroException ex) {
      Console.WriteLine("Error: " + ex.Message);
    }
  }

  static int Divide(int x, int y) {
    return x / y; // Step into this method
  }

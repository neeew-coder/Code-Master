using System;

class Book {
  public string Title;
  public string Author;

  public Book(string title, string author) {
    Title = title;
    Author = author;
  }

  public void Display() {
    Console.WriteLine($"\"{Title}\" by {Author}");
  }
}

class Program {
  static void Main() {
    Book b = new Book("Clean Code", "Robert C. Martin");
    b.Display();
  }

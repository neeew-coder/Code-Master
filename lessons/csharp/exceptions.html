// ExceptionDemo.cs
using System;

class Program {
  static void Main() {
    try {
      int[] numbers = { 1, 2, 3 };
      Console.WriteLine(numbers[5]); // Index out of range
    } catch (IndexOutOfRangeException ex) {
      Console.WriteLine("Error: " + ex.Message);
    } finally {
      Console.WriteLine("Execution completed.");
    }
  }
}

// FileDemo.cs
using System;
using System.IO;

class Program {
  static void Main() {
    string path = "sample.txt";

    // Write to file
    File.WriteAllText(path, "Hello, Mark!");

    // Read from file
    string content = File.ReadAllText(path);
    Console.WriteLine("File content: " + content);

    // Append to file
    File.AppendAllText(path, "\nWelcome to CodeMaster!");

    // Read again
    Console.WriteLine(File.ReadAllText(path));
  }
}

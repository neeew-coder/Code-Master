// OverloadingDemo.cs
using System;

class OverloadingDemo {
  static void Main() {
    Greet();
    Greet("Mark");
    Greet("Mark", 3);
  }

  static void Greet() {
    Console.WriteLine("Hello!");
  }

  static void Greet(string name) {
    Console.WriteLine("Hello, " + name + "!");
  }

  static void Greet(string name, int times) {
    for (int i = 0; i < times; i++) {
      Console.WriteLine("Hello, " + name + "!");
    }
  }
}

// ClassMembersDemo.cs
using System;

class Person {
  private string name;
  public int Age { get; set; }

  public Person(string name, int age) {
    this.name = name;
    Age = age;
  }

  public void Introduce() {
    Console.WriteLine($"Hi, I'm {name} and I'm {Age} years old.");
  }
}

class Program {
  static void Main() {
    Person p = new Person("Mark", 30);
    p.Introduce();
  }
}

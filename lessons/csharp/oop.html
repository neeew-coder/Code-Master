// OOPDemo.cs
using System;

class Animal {
  public virtual void Speak() {
    Console.WriteLine("Animal speaks");
  }
}

class Dog : Animal {
  public override void Speak() {
    Console.WriteLine("Dog barks");
  }
}

class Program {
  static void Main() {
    Animal myAnimal = new Dog(); // Polymorphism
    myAnimal.Speak();            // Output: Dog barks
  }
}

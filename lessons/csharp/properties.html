// PropertiesDemo.cs
using System;

class Student {
  private string name;
  public string Name {
    get { return name; }
    set {
      if (!string.IsNullOrWhiteSpace(value)) {
        name = value;
      }
    }
  }

  public int Age { get; private set; }

  public Student(string name, int age) {
    Name = name;
    Age = age;
  }
}

class Program {
  static void Main() {
    Student s = new Student("Mark", 20);
    Console.WriteLine($"Name: {s.Name}, Age: {s.Age}");
  }
}

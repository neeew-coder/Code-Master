// WhileLoopDemo.java
public class WhileLoopDemo {
  public static void main(String[] args) {
    int count = 1;

    while (count <= 5) {
      System.out.println("Count: " + count);
      count++;
    }

    int num = 0;
    do {
      System.out.println("Do-While: " + num);
      num++;
    } while (num < 3);
  }
}

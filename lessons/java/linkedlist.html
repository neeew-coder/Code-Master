import java.util.LinkedList;

public class LinkedListDemo {
  public static void main(String[] args) {
    LinkedList tasks = new LinkedList<>();
    tasks.add("Write");
    tasks.add("Test");
    tasks.addFirst("Plan");
    tasks.addLast("Deploy");

    tasks.remove("Test");
    tasks.removeFirst();

    for (String task : tasks) {
      System.out.println(task);
    }
  }
}

// HashSetDemo.java
import java.util.HashSet;

public class HashSetDemo {
  public static void main(String[] args) {
    HashSet emails = new HashSet<>();
    emails.add("user1@example.com");
    emails.add("user2@example.com");
    emails.add("user1@example.com"); // Duplicate ignored

    System.out.println("Emails: " + emails);
    System.out.println("Contains user2? " + emails.contains("user2@example.com"));

    emails.remove("user2@example.com");
    System.out.println("After removal: " + emails);
  }
}

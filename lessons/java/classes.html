class Car {
  String model;
  int year;

    this.model = model;
    this.year = year;
  }

  void displayInfo() {

  }
}

public class ClassDemo {
  public static void main(String[] args) {
    Car myCar = new Car("Toyota", 2022);
    myCar.displayInfo();
  }

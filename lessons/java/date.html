Java provides multiple APIs for handling date and time. The legacy Date and Calendar classes offer basic functionality, while the modern java.time package (introduced in Java 8) provides immutable, thread-safe classes like LocalDate, LocalTime, and LocalDateTime. Formatting and parsing are handled via DateTimeFormatter, enabling clean, localized output.

// DateDemo.java
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;

public class DateDemo {
  public static void main(String[] args) {
    LocalDate today = LocalDate.now();
    LocalDateTime now = LocalDateTime.now();

    DateTimeFormatter formatter = DateTimeFormatter.ofPattern("MMMM dd, yyyy HH:mm");

    System.out.println("Today's date: " + today);
    System.out.println("Current time: " + now.format(formatter));
  }
}
